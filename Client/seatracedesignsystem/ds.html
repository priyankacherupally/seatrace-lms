02 Custom Palette Colors
Add any colour beyond the standard six. Option A requires manual tokens; Option B uses augmentColor to auto-calculate light, dark, and contrastText from just a main.
// Option A — manual tokens
const theme = createTheme({
  palette: {
    ocean: { main: '#0077B6', light: '#00B4D8', dark: '#03045E', contrastText: '#fff' },
  },
});

// Option B — augmentColor (auto-calculates light / dark / contrastText)
let theme = createTheme();
theme = createTheme(theme, {
  palette: {
    ocean: theme.palette.augmentColor({
      color: { main: '#0077B6' },
      name: 'ocean',
    }),
  },
});
03 Palette Utility Functions
alpha()
Apply opacity to any colour value
alpha('#5189F3', 0.5)
// rgba(81,137,243,0.5)
getContrastRatio()
Check WCAG contrast between two colours
getContrastRatio('#5189F3','#fff')
// e.g. 3.2 — AA needs ≥ 4.5
import { alpha, getContrastRatio } from '@mui/material/styles';

// Opacity
const translucent = alpha('#5189F3', 0.5);   // rgba(81,137,243,0.5)

// WCAG check — use contrastThreshold: 4.5 in theme for AA
const ratio = getContrastRatio('#5189F3', '#fff');
const textColor = ratio > 4.5 ? '#fff' : '#111';

06 Spacing System
Number (Seatrace)
spacing: 4
spacing(1) → 4px
Function
f => 0.25 * f + rem
Bootstrap approach
Array
[0,4,8,16,32,64]
Explicit scale
// Seatrace uses 4px unit
theme.spacing(1)       // → '4px'
theme.spacing(2)       // → '8px'
theme.spacing(4)       // → '16px'
theme.spacing(1, 2)   // → '4px 8px'  (shorthand)
theme.spacing(1, 'auto') // → '4px auto'
07 Component Customization Methods
Method 1 Default Props — change prop defaults globally 
components: {
  MuiButton:     { defaultProps: { disableElevation: true } },
  MuiTextField:  { defaultProps: { size: 'small', variant: 'outlined' } },
  MuiButtonBase: { defaultProps: { disableRipple: true } },
}
Method 2 Style Overrides — override CSS per slot 
MuiButton: {
  styleOverrides: {
    root: { borderRadius: 4, fontWeight: 600 },
    containedPrimary: {
      boxShadow: 'none',
      '&:hover': { backgroundColor: '#70A0FB', boxShadow: 'none' },
      '&:active': { backgroundColor: '#326DDD' },
    },
    outlined: { borderColor: '#5189F3' },
  },
}
Method 3 Conditional Variants — styles triggered by prop values 
MuiButton: {
  styleOverrides: {
    root: {
      variants: [
        { props: { variant: 'dashed' },
          style: { border: '2px dashed #5189F3' } },
        { props: { variant: 'contained', color: 'primary' },
          style: { boxShadow: 'none' } },
      ],
    },
  },
}
Method 4 Global Theme Variables — cascades to all components automatically 
createTheme({
  typography: { button: { fontSize: '0.875rem' } },
  shape:      { borderRadius: 8 },
  spacing:    4,
});
